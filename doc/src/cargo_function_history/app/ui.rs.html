16pub fn draw(rect: &mut Frame, app: &mut App) {
17    let size = rect.area();
18    // check if we have enough space to draw
19    if size.width < 10 || size.height < 10 {
20        panic!("Not enough space to draw");
21    }
22    let main_window = draw_main();
23    let mut whole_chunks = Layout::default()
24        .direction(Direction::Vertical)
25        .constraints([Constraint::Length(1), Constraint::Length(size.height - 1)].as_ref())
26        .split(size)[1];
27    rect.render_widget(main_window, size);
28    whole_chunks = Layout::default()
29        .direction(Direction::Horizontal)
30        .constraints(
31            [
32                Constraint::Length(1),
33                Constraint::Length(whole_chunks.width - 2),
34                Constraint::Length(1),
35            ]
36            .as_ref(),
37        )
38        .split(whole_chunks)[1];
39    let body_chunks = Layout::default()
40        .direction(Direction::Vertical)
41        .constraints(
42            [
43                Constraint::Length(whole_chunks.height - 4),
44                Constraint::Length(2),
45                Constraint::Length(2),
46            ]
47            .as_ref(),
48        )
49        .split(whole_chunks);
50    app.get_result();
51    draw_body(
52        app,
53        *body_chunks.first().expect("could not get area to draw"),
54        rect,
55    );
56    rect.render_widget(
57        &app.input_buffer,
58        *body_chunks.get(1).expect("could not get area to draw"),
59    );
60    let status = draw_status(app.status());
61    rect.render_widget(
62        status,
63        *body_chunks.get(2).expect("could not get area to draw"),
64    );
65}

67fn draw_body(app: &mut App, mut pos: Rect, frame: &mut Frame) {
68    let top = match &app.cmd_output {
69        CommandResult::History(history) => {
70            let metadata = history.get_metadata();
71            let metadata = BTreeMap::from_iter(metadata.iter());
72            let metadata: Vec<Line> = metadata
73                .iter()
74                .map(|x| Line::from(format!("{}: {}\n", x.0, x.1)))
75                .collect();
76            Some(
77                Paragraph::new(metadata)
78                    .style(Style::default().fg(Color::LightCyan))
79                    .block(Block::default().style(Style::default().fg(Color::White))),
80            )
81        }
82        _ => None,
83    };
84    let tick_text: Vec<Line> = match &app.cmd_output {
85        CommandResult::None => match app.status {
86            Status::Loading => vec![Line::from(format!(
87                "Loading{}",
88                ".".repeat(
89                    ((std::time::SystemTime::now()
90                        .duration_since(std::time::UNIX_EPOCH)
91                        .expect("Time went backwards")
92                        .as_millis()
93                        / 100)
94                        % 4) as usize
95                )
96            ))],
97            _ => vec![Line::from(
98                "Please enter some commands to search for a function.",
99            )],
100        },
101        a => a
102            .to_string()
103            .split('\n')
104            .map(|s| Line::from(format!("{s}\n")))
105            .collect(),
106    };
107    app.scroll_state = app.scroll_state.content_length(tick_text.len());
108    let body = Paragraph::new(tick_text)
109        .style(Style::default().fg(Color::LightCyan))
110        .scroll(app.scroll_pos)
111        .block(
112            Block::default()
113                .borders(Borders::BOTTOM)
114                .style(Style::default().fg(Color::White)),
115        );
116    if let Some(top) = top {
117        let mut top_pos = pos;
118        top_pos.height = 4;
119        pos.height -= 4;
120        pos.y += 4;
121        frame.render_widget(top, top_pos);
122    }
123    app.body_height = pos.height;
124    frame.render_widget(body, pos);
125
126    frame.render_stateful_widget(
127        Scrollbar::default()
128            .orientation(ScrollbarOrientation::VerticalRight)
129            .begin_symbol(Some("↑"))
130            .end_symbol(Some("↓")),
131        pos,
132        &mut app.scroll_state,
133    )
134}

136fn draw_main<'a>() -> Block<'a> {
137    Block::default()
138        .borders(Borders::ALL)
139        .title(" Cargo Function History ")
140        .border_style(Style::default().fg(Color::White))
141        .title_alignment(Alignment::Center)
142        .style(Style::default().fg(Color::White))
143}

145fn draw_status<'a>(status: &Status) -> Paragraph<'a> {
146    Paragraph::new(vec![Line::from(Span::raw(status.to_string()))])
147        .style(Style::default().fg(Color::LightCyan))
148        .block(
149            Block::default()
150                .borders(Borders::TOP)
151                .borders(Borders::BOTTOM)
152                .style(Style::default().fg(Color::White)),
153        )
