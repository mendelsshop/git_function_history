189    fn draw_history(history: &mut FunctionHistory, ctx: &egui::Context) {
190        // split the screen top and bottom into two parts, leave small part for the left arrow commit hash and right arrow and the rest for the content
191        // create a 3 line header
192        TopBottomPanel::top("control history").show(ctx, |ui| {
193            ui.set_height(2.0);
194            ui.horizontal(|ui| {
195                let mut max = ui.available_width();
196                let l_resp = match history.get_move_direction() {
197                    Directions::Forward => {
198                        ui.add_sized(Vec2::new(2.0, 2.0), Button::new("<-").sense(Sense::hover()));
199                        None
200                    }
201                    _ => Some(
202                        // add a left arrow button that is disabled
203                        ui.add_sized(Vec2::new(2.0, 2.0), Button::new("<-")),
204                    ),
205                };
206                max -= ui.available_width();
207                ui.add_sized(
208                    Vec2::new(ui.available_width() - max, 2.0),
209                    Label::new(format!(
210                        "{}\n{}",
211                        history
212                            .get_metadata()
213                            .get("commit hash")
214                            .map_or("could not retrieve commit hash", |x| x.as_str()),
215                        history
216                            .get_metadata()
217                            .get("date")
218                            .map_or("could not retieve date", |x| x.as_str()),
219                    )),
220                );
221
222                let r_resp = match history.get_move_direction() {
223                    Directions::Back => {
224                        ui.add_sized(Vec2::new(2.0, 2.0), Button::new("->").sense(Sense::hover()));
225                        None
226                    }
227                    _ => {
228                        // add a right arrow button that is disabled
229                        Some(ui.add_sized(Vec2::new(2.0, 2.0), Button::new("->")))
230                    }
231                };
232
233                if let Some(r_resp) = r_resp {
234                    if r_resp.clicked() {
235                        history.move_forward();
236                    }
237                }
238                if let Some(l_resp) = l_resp {
239                    if l_resp.clicked() {
240                        history.move_back();
241                    }
242                }
243            });
244        });
245        if let Some(x) = history.get_mut_commit() {
246            Self::draw_commit(x, ctx, false)
247        } else {
248            TopBottomPanel::top("no_commit_found").show(ctx, |ui| {
249                ui.add(Label::new("no commit found"));
250            });
251        }
252    }
253    fn handle_many_filter(
254        inputs: &mut HashMap<String, (bool, String)>,
255        ui: &mut egui::Ui,
256        max: f32,
257        next_field: &mut String,
258        specific: &mut Option<String>,
259        languages: &[String],
260        filters: &function_grep::filter::FilterType<'static>,
261    ) -> Result<Option<PLFilter>, String> {
262        // TODO: cleanup
263        inputs.iter_mut().for_each(|(desc, field)| {
264            ui.horizontal(|ui| {
265                ui.set_min_width(4.0);
266                ui.set_max_width(max);
267                ui.label(desc.to_string());
268                ui.add(TextEdit::singleline(&mut field.1));
269                field.0 = ui.add(Button::new("-")).clicked();
270            });
271        });
272        ui.add(TextEdit::singleline(next_field));
273        let resp = ui.add(Button::new("add field"));
274        if resp.clicked() {
275            inputs.insert(next_field.to_string(), (false, String::new()));
276            next_field.clear();
277        }
278        inputs.retain(|_, input| !input.0);
279        egui::ComboBox::from_id_salt("filter_language_chooser")
280            .selected_text("Language")
281            .show_ui(ui, |ui| {
282                ui.selectable_value(specific, None, "All");
283                languages.iter().for_each(|name| {
284                    ui.selectable_value(specific, Some(name.to_string()), name);
285                });
286            });
287        if let Some(language) = specific {
288            {
289                match filters.specific(language) {
290                    Some(filter @ function_grep::filter::SingleOrMany::Single(single_filter)) => {
291                        Ok(Some(instantiate_single_filter_state_inner(
292                            single_filter,
293                            filter,
294                        )))
295                    }
296                    _ => Err(format!(
297                        "Filter {filters} does not support Language {language}."
298                    )),
299                }
300            }
301        } else {
302            Ok(None)
303        }
304    }

829fn instantiate_filter(
830    filter: &function_grep::filter::FilterType<'static>,
831    input: impl Iterator<Item = (impl fmt::Display, impl fmt::Display)>,
832) -> Result<function_grep::filter::InstantiatedFilterType, String> {

834        &input
835            .map(|v| format!("{}: {}", v.0, v.1))
836            .collect_vec()
837            .join(" "),
838    )

852fn initialize_filter_state(
853    filter: function_grep::filter::FilterType<'static>,
854) -> HistoryFilterType {
855    match filter {
856        function_grep::filter::FilterType::All(all_filter) => {
857            instantiate_single_filter_state(all_filter, filter)
858        }
859        function_grep::filter::FilterType::Single(single_filter) => {
860            instantiate_single_filter_state(single_filter, filter)
861        }
862
863        function_grep::filter::SingleOrMany::Many(ref filters) => {
864            let single_filter = filters
865                .filters
866                .keys()
867                .next()
868                .filter(|_| filters.filters.len() == 1)
869                .and_then(|k| filter.specific(k));
870            match single_filter {
871                Some(filter @ function_grep::filter::SingleOrMany::Single(single_filter)) => {
872                    instantiate_single_filter_state(single_filter, filter)
873                }
874                _ => {
875                    let collect_vec = filters.filters.keys().cloned().collect_vec();
876
877                    HistoryFilterType::PL(types::PLFilter::Many(
878                        HashMap::new(),
879                        filter,
880                        collect_vec,
881                        None,
882                        String::new(),
883                    ))
884                }
885            }
886        }
887    }
888}

890fn instantiate_single_filter_state<T>(
891    allfilter: &dyn function_grep::filter::Filter<Supports = T>,
892    filter: function_grep::filter::FilterType<'static>,
893) -> HistoryFilterType
894where
895    function_grep::SupportedLanguages: From<T>,
896{
897    HistoryFilterType::PL(instantiate_single_filter_state_inner(allfilter, filter))
898}

900fn instantiate_single_filter_state_inner<T>(
901    allfilter: &dyn function_grep::filter::Filter<Supports = T>,
902    filter: function_grep::filter::FilterType<'static>,
903) -> PLFilter
904where
905    function_grep::SupportedLanguages: From<T>,
906{
907    types::PLFilter::Single(
908        allfilter
909            .attributes()
910            .into_iter()
911            .map(|(attr, kind)| (attr.to_string(), kind.to_string()))
912            .collect(),
913        filter,
914    )
