225pub type FilterType<'a> =
226    SingleOrMany<&'a dyn Filter<Supports = All>, &'a dyn Filter<Supports = Language>>;
227pub type InstantiatedFilterType =
228    SingleOrMany<InstantiatedFilter<All>, InstantiatedFilter<Language>>;
229#[derive(Debug, PartialEq, Eq)]
230pub enum SingleOrMany<A: Info<Supported = All>, M: Info<Supported = Language>> {
231    All(A),
232    Many(Many<M>),
233    Single(M),
234}
235impl<A: Info<Supported = All>, M: Info<Supported = Language>> SingleOrMany<A, M> {
236    #[must_use]
237    pub fn filter_name(&self) -> String {
238        match self {
239            Self::All(f) => f.filter_name(),
240            Self::Many(many) => many.name.clone(),
241            Self::Single(s) => s.filter_name(),
242        }
243    }
244}
245impl<A: Info<Supported = All>, M: Info<Supported = Language>> Display for SingleOrMany<A, M> {
246    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
247        write!(f, "{}", self.filter_name())
248    }
249}
250impl FilterType<'_> {
251    #[must_use]
252    pub fn specific(&self, s: &str) -> Option<Self> {
253        match self {
254            Self::All(_) | Self::Single(_) => None,
255            Self::Many(m) => m.filters.get(s).map(|x| Self::Single(*x)),
256        }
257    }
258
259    pub fn to_filter(&self, s: &str) -> Result<InstantiatedFilterType, String> {
260        match self {
261            SingleOrMany::All(a) => a.to_filter(s).map(SingleOrMany::All),
262            SingleOrMany::Single(a) => a.to_filter(s).map(SingleOrMany::Single),
263            SingleOrMany::Many(Many { name, filters }) => filters
264                .iter()
265                .map(|(name, f)| f.to_filter(s).map(|f| (name.clone(), f)))
266                .collect::<Result<HashMap<_, _>, _>>()
267                .map(|filters| {
268                    SingleOrMany::Many(Many {
269                        name: name.clone(),
270                        filters,
271                    })
272                }),
273        }
274    }

353impl<'a> Filters<'a> {
354    pub fn add_filter(&mut self, filter: impl Into<FilterType<'a>>) -> Result<(), String> {
355        let filter: FilterType<'a> = filter.into();
356        let name = filter.filter_name();
357        if let Some(filters) = self.filters.get_mut(&name) {
358            match filters {
359                SingleOrMany::All(_) => Err("cannot add with or to an all filter".to_string()),
360                _ if matches!(filter, SingleOrMany::All(_)) => {
361                    Err("cannot add with or to an all filter".to_string())
362                }
363                SingleOrMany::Many(filter1) => match filter {
364                    SingleOrMany::All(_) => unreachable!(),
365                    SingleOrMany::Many(filter2) => try_extend_filter(filter1, filter2),
366                    SingleOrMany::Single(filter2) => try_add_filter(filter1, filter2),
367                },
368                SingleOrMany::Single(filter1) => match filter {
369                    SingleOrMany::All(_) => unreachable!(),
370                    SingleOrMany::Many(mut filter2) => {
371                        try_add_filter(&mut filter2, *filter1)?;
372                        *filters = SingleOrMany::Many(filter2);
373                        Ok(())
374                    }
375                    SingleOrMany::Single(filter2) if filter1.supports() == filter2.supports() => {
376                        Err(format!(
377                            "cannot add duplicate filter {name} for {}",
378                            filter1.supports()
379                        ))
380                    }
381                    SingleOrMany::Single(filter2) => {
382                        *filters = SingleOrMany::Many(Many {
383                            name,
384                            filters: HashMap::from([
385                                (filter1.supports().0, *filter1),
386                                (filter2.supports().0, filter2),
387                            ]),
388                        });
389                        Ok(())
390                    }
391                },
392            }
393        } else {
394            self.filters.insert(name, filter);
395            Ok(())
396        }
397    }
398    #[must_use]
399    pub fn get_filter(&self, name: &str) -> Option<&FilterType<'a>> {
400        self.filters.get(name)
401    }
402}
403
404fn try_add_filter<'a>(
405    filters: &mut Many<&'a dyn Filter<Supports = Language>>,
406    filter: &'a dyn Filter<Supports = Language>,
407) -> Result<(), String> {
408    let mut status = Ok(());
409    filters
410        .filters
411        .entry(filter.supports().0)
412        .and_modify(|_| {
413            status = Err(format!(
414                "cannot add duplicate filter {} for {}",
415                filter.filter_name(),
416                filter.supports()
417            ));
418        })
419        .or_insert(filter);
420    status
421}
422
423fn try_extend_filter<'a>(
424    filters: &mut Many<&'a dyn Filter<Supports = Language>>,
425    new_filters: Many<&'a dyn Filter<Supports = Language>>,
426) -> Result<(), String> {
427    new_filters
428        .filters
429        .into_values()
430        .try_for_each(|filter| try_add_filter(filters, filter))
