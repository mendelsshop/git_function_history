1use function_grep::filter::FilterType;
2use std::{collections::HashMap, fmt};
3
4pub enum HistoryFilterType {
5    Date(String),
6    DateRange(String, String),
7    CommitHash(String),
8    FileAbsolute(String),
9    FileRelative(String),
10    Directory(String),
11    // if filter type is a many and it has more than on filter
12    // 1. if you can pick which language to use
13    // 2. or you can add or remove a field to apply to the filter
14    PL(PLFilter),
15    None,
16}
17pub enum PLFilter {
18    Single(HashMap<String, String>, FilterType<'static>),
19    Many(
20        // if the field should be deleted
21        HashMap<String, (bool, String)>,
22        FilterType<'static>,
23        Vec<String>,
24        // choose a sepcific language that supports this filter
25        Option<String>,
26        // next field
27        String,
28    ),
29}
30impl PLFilter {
31    fn filter_name(&self) -> String {
32        match self {
33            PLFilter::Single(_, single_or_many) => single_or_many.filter_name(),
34            PLFilter::Many(_, filter, _, _, _) => filter.filter_name(),
35        }
36    }
37}
38
39impl PartialEq for HistoryFilterType {
40    fn eq(&self, other: &Self) -> bool {
41        match (self, other) {
42            (Self::Date(l0), Self::Date(r0)) => l0 == r0,
43            (Self::DateRange(l0, l1), Self::DateRange(r0, r1)) => l0 == r0 && l1 == r1,
44            (Self::CommitHash(l0), Self::CommitHash(r0)) => l0 == r0,
45            (Self::FileAbsolute(l0), Self::FileAbsolute(r0)) => l0 == r0,
46            (Self::FileRelative(l0), Self::FileRelative(r0)) => l0 == r0,
47            (Self::Directory(l0), Self::Directory(r0)) => l0 == r0,
48            (Self::PL(l0), Self::PL(r0)) => l0.filter_name() == r0.filter_name(),
49            _ => core::mem::discriminant(self) == core::mem::discriminant(other),
50        }
51    }
52}

54impl Eq for HistoryFilterType {}

56impl fmt::Debug for HistoryFilterType {
57    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
58        match self {
59            Self::Date(arg0) => f.debug_tuple("Date").field(arg0).finish(),

61                f.debug_tuple("DateRange").field(arg0).field(arg1).finish()
62            }
63            Self::CommitHash(arg0) => f.debug_tuple("CommitHash").field(arg0).finish(),
64            Self::FileAbsolute(arg0) => f.debug_tuple("FileAbsolute").field(arg0).finish(),
65            Self::FileRelative(arg0) => f.debug_tuple("FileRelative").field(arg0).finish(),
66            Self::Directory(arg0) => f.debug_tuple("Directory").field(arg0).finish(),
67            Self::PL(filter) => f.debug_tuple("PL").field(&filter.filter_name()).finish(),
68            Self::None => write!(f, "None"),
69        }
70    }
71}
