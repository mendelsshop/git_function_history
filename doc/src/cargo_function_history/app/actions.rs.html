9/// We define all available action
10#[derive(Debug, Clone, Copy, Eq, PartialEq)]
11pub enum Action {
12    Quit,
13    TextEdit,
14    ScrollUp,
15    ScrollDown,
16    BackCommit,
17    ForwardCommit,
18    BackFile,
19    ForwardFile,
20}
21
22impl Action {
23    /// All available actions
24    pub fn iterator() -> Iter<'static, Action> {
25        static ACTIONS: [Action; 8] = [
26            Action::Quit,
27            Action::TextEdit,
28            Action::ScrollUp,
29            Action::ScrollDown,
30            Action::BackCommit,
31            Action::ForwardCommit,
32            Action::BackFile,
33            Action::ForwardFile,
34        ];
35        ACTIONS.iter()
36    }
37
38    /// List of key associated to action
39    pub fn keys(&self) -> &[Key] {
40        match self {
41            Action::Quit => &[Key::Ctrl('c'), Key::Char('q')],
42            Action::TextEdit => &[Key::Char(':'), Key::Shift(':')],
43            Action::ScrollUp => &[Key::Up, Key::Char('k')],
44            Action::ScrollDown => &[Key::Down, Key::Char('j')],
45            Action::BackCommit => &[Key::Left, Key::Char('h')],
46            Action::ForwardCommit => &[Key::Right, Key::Char('l')],
47            Action::BackFile => &[Key::Shiftleft, Key::Char('H'), Key::Shift('h')],
48            Action::ForwardFile => &[Key::Shiftright, Key::Char('L'), Key::Shift('l')],
49        }
50    }
51}

73/// The application should have some contextual actions.
74#[derive(Default, Debug, Clone)]
75pub struct Actions(Vec<Action>);
76
77impl Actions {
78    /// Given a key, find the corresponding action
79    pub fn find(&self, key: Key) -> Option<&Action> {
80        log::debug!("{}", key);
81        Action::iterator()
82            .filter(|action| self.0.contains(action))
83            .find(|action| action.keys().contains(&key))
84    }
85
86    /// Get contextual actions.
87    /// (just for building a help view)
88    pub fn actions(&self) -> &[Action] {
89        self.0.as_slice()
90    }
91}
92
93impl From<Vec<Action>> for Actions {
94    /// Build contextual action
95    ///
96    /// # Panics
97    ///
98    /// If two actions have same key
99    fn from(actions: Vec<Action>) -> Self {
100        // Check key unicity
101        let mut map: HashMap<Key, Vec<Action>> = HashMap::new();
102        for action in actions.iter() {
103            for key in action.keys().iter() {
104                match map.get_mut(key) {
105                    Some(vec) => vec.push(*action),
106                    None => {
107                        map.insert(*key, vec![*action]);
108                    }
109                }
110            }
111        }
112        let errors = map
113            .iter()
114            .filter(|(_, actions)| actions.len() > 1) // at least two actions share same shortcut
115            .map(|(key, actions)| {
116                let actions = actions
117                    .iter()
118                    .map(Action::to_string)
119                    .collect::<Vec<_>>()
120                    .join(", ");
121                format!("Conflict key {key} with actions {actions}")
122            })
123            .collect::<Vec<_>>();
124        if !errors.is_empty() {
125            panic!("{}", errors.join("; "))
126        }
127
128        // Ok, we can create contextual actions
129        Self(actions)
130    }
